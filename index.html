function buildBnav() { const bn = document.getElementById('bnav'); bn.innerHTML = ''; bnavDefs.forEach((item, i) => { const navDef = navDefs.find(n => n.id === item.id); const locked = navDef?.premium && APP.tier !== 'premium'; const div = document.createElement('div'); div.className = 'bnav-item' + (i === 0 ? ' active' : ''); div.innerHTML = `
${item.icon}
${item.label}
`; div.onclick = () => { if (locked) { toast('🔒 Requires Premium — upgrade in Plan & Features'); return; } document.querySelectorAll('.bnav-item').forEach(b => b.classList.remove('active')); div.classList.add('active'); showPage(item.id, null); // Pass null for navEl for bottom nav }; bn.appendChild(div); }); }
